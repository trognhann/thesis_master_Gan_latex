\section{Bộ Dữ liệu}
\label{sec:dataset}

\subsection{Dữ liệu Ảnh Thực (Photo)}
\label{subsec:photo_data}

Tập dữ liệu ảnh thực dùng cho huấn luyện bao gồm ảnh chân dung từ nhiều nguồn:

\begin{itemize}
    \item Ảnh chân dung đa dạng về giới tính, độ tuổi, sắc tộc và ánh sáng.
    \item Ảnh được lọc để loại bỏ các ảnh quá mờ, quá tối, hoặc không có khuôn mặt nổi bật.
    \item Kích thước mỗi ảnh được resize về $256 \times 256$ (landscape mode) hoặc $512 \times 512$ (face mode) trước khi đưa vào mô hình.
\end{itemize}

Tiền xử lý tạo hai tập con phục vụ các thành phần loss:
\begin{itemize}
    \item \texttt{seg\_train\_5-0.8-50/}: Ảnh đã qua Felzenszwalb superpixel segmentation ($\sigma=5$, $k=0.8$, $\text{min\_size}=50$) — phục vụ Region Smoothing Loss.
\end{itemize}

\subsection{Dữ liệu Ảnh Anime (Style)}
\label{subsec:anime_data}

Dữ liệu phong cách anime được trích xuất từ phim hoạt hình của các đạo diễn nổi tiếng:
\begin{itemize}
    \item \textbf{Hayao Miyazaki:} Các frame từ phim \textit{Spirited Away}, \textit{My Neighbor Totoro} — phong cách warm, giàu màu sắc tự nhiên, đường nét mềm mại.
    \item \textbf{Makoto Shinkai:} Các frame từ \textit{Your Name}, \textit{Weathering With You} — phong cách lạnh, ánh sáng cinematic, chi tiết mịn.
    \item \textbf{Phong cách khác:} Thêm các phong cách anime đa dạng để mở rộng khả năng chuyển đổi.
\end{itemize}

Mỗi tập anime đi kèm tập \texttt{smooth/} được tạo bằng Edge Smoothing — làm mờ nhẹ vùng biên trong ảnh anime, phục vụ Discriminator tập trung vào đường nét sắc nét.

\begin{figure}[H]
    \centering
    \includegraphics[width=0.85\textwidth]{figChap3/preTrainGLoss.png}
    \caption{Đường cong hội tụ Pre-train G\_loss trong giai đoạn pre-training (5 epoch đầu). Loss giảm nhanh từ $\approx 0.9$ xuống $\approx 0.15$ trong 500 steps đầu, sau đó tiếp tục giảm dần và ổn định ở $\approx 0.12$ -- cho thấy Generator đã học được biểu diễn nội dung tốt trước khi kích hoạt adversarial training.}

\subsection{Giai đoạn Huấn luyện Đầy đủ}
\label{subsec:full_training_results}

\subsubsection{Tổng G\_loss và D\_loss}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{figChap3/Dloss_Gloss.png}
    \caption{Đường cong G\_loss (xanh lam) và D\_loss (cam) qua 3.500 steps trong giai đoạn huấn luyện đầy đủ. G\_loss giảm mạnh từ $\approx 44$ xuống $\approx 8$ trong 500 steps đầu, sau đó ổn định và dao động trong vùng $[7, 10]$. D\_loss duy trì ổn định gần 0 trong suốt quá trình -- biểu hiện điển hình của cân bằng GAN khi Discriminator đủ mạnh.}
    \label{fig:gloss_dloss}
\end{figure}

Đặc điểm đáng chú ý từ Hình~\ref{fig:gloss_dloss}:
\begin{itemize}
    \item \textbf{G\_loss bắt đầu rất cao ($\approx 44$)} do lần đầu kích hoạt adversarial loss và revision loss -- Generator cần thích nghi với tín hiệu từ Discriminator.
    \item \textbf{Giảm nhanh trong 500 steps đầu:} Generator hội tụ nhanh về vùng cân bằng nhờ pre-training đã cho khởi điểm feature tốt.
    \item \textbf{Dao động ổn định sau đó:} Dao động của G\_loss trong vùng $[7,10]$ là bình thường và kỳ vọng trong GAN training -- phản ánh quá trình ``cạnh tranh'' liên tục giữa Generator và Discriminator.
    \item \textbf{D\_loss rất nhỏ và ổn định:} Discriminator học phân biệt hiệu quả từ sớm và duy trì ổn định, không bị collapse.
\end{itemize}

\subsubsection{Các thành phần Loss phụ}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{figChap3/orther_loss.png}
    \caption{Đường cong các thành phần loss phụ: color\_loss (xanh lam, $\approx 0.6$), con\_loss (cam, $\approx 0.15$) và sty\_loss (xanh lá, dao động $[2.5, 4.0]$) qua 3.500 steps. Style loss dao động lớn phản ánh bản chất khó đo lường của phong cách anime -- đây là hành vi bình thường.}
    \label{fig:other_loss}
\end{figure}

Phân tích các thành phần loss trong Hình~\ref{fig:other_loss}:
\begin{itemize}
    \item \textbf{con\_loss (Content Loss, cam):} Rất nhỏ và ổn định ($\approx 0.15$) -- Generator bảo toàn tốt nội dung ảnh gốc trong suốt quá trình huấn luyện.
    \item \textbf{color\_loss (Lab Color Loss, xanh lam):} Dao động nhỏ quanh $0.6$ -- màu sắc ảnh đầu ra khá nhất quán với ảnh gốc.
    \item \textbf{sty\_loss (Style Loss, xanh lá):} Dao động lớn trong $[2.5, 4.0]$ -- phản ánh bản chất phức tạp của học phong cách: có thời điểm Generator tạo ảnh ``rất anime'' (loss nhỏ) và có lúc ``ít anime hơn'' khi cân bằng với các loss khác.
\end{itemize}

\section{Thảo luận và Phân tích}
\label{sec:discussion}

\subsection{Điểm mạnh của Hệ thống}
\label{subsec:strengths}

\begin{itemize}
    \item \textbf{Chất lượng ảnh đầu ra cao:} FID 58.6 (Face Mode) cho thấy phân phối ảnh sinh gần với ảnh anime thật hơn so với các phương pháp baseline.
    \item \textbf{Hiệu năng tốt:} 42 FPS ở $256\times256$ (GPU) và model size chỉ 9.1 MB ONNX phù hợp cho deployment thực tế.
    \item \textbf{Kiến trúc Double-Tail hiệu quả:} Hai nhánh decoder chuyên biệt học các đặc trưng bổ sung nhau -- Support Tail cho đường nét, Main Tail cho chi tiết mịn.
    \item \textbf{LADE normalization:} Tự thích ứng tham số chuẩn hóa linh hoạt hơn Instance Norm cố định.
    \item \textbf{Pipeline tích hợp:} Từ phát hiện mặt đến chuyển đổi và hiển thị trong một luồng seamless.
\end{itemize}

\subsection{Hạn chế và Định hướng Cải thiện}
\label{subsec:limitations}

\begin{itemize}
    \item \textbf{Phụ thuộc chất lượng dữ liệu:} Chất lượng và đa dạng của tập anime style ảnh hưởng trực tiếp đến phong cách học được.
    \item \textbf{Khuôn mặt nghiêng lớn:} RetinaFace MobileNet 0.25 giảm hiệu năng đáng kể ở góc nghiêng $>60°$; cần face alignment để xử lý tốt hơn.
    \item \textbf{Chưa hỗ trợ video real-time:} Pipeline hiện tại xử lý frame-by-frame độc lập, chưa tối ưu cho video (flickering giữa các frame).
    \item \textbf{Background bị bỏ qua ở Face Mode:} Người dùng phải tự ghép mặt đã anime hóa vào nền gốc nếu muốn kết quả toàn ảnh.
\end{itemize}

\section*{Tổng kết chương}

Chương này đã trình bày đầy đủ các kết quả thực nghiệm của hệ thống AnimeGANv3 DTGAN:

\begin{itemize}
    \item \textbf{Huấn luyện hội tụ tốt:} Pre-train G\_loss giảm từ 0.9 xuống 0.12; G\_loss/D\_loss ổn định sau $\approx$500 steps adversarial training; các loss thành phần duy trì trong vùng kỳ vọng.
    \item \textbf{Vượt trội định lượng:} FID 58.6 và LPIPS 0.384 (Face Mode) -- tốt nhất trong số các phương pháp so sánh; tốc độ 42 FPS tại $256\times256$ GPU.
    \item \textbf{Định tính đa dạng phong cách:} Hệ thống chuyển đổi thành công nhiều phong cách anime khác nhau, bảo toàn nhận dạng khuôn mặt và biểu cảm.
    \item \textbf{Face Extraction hiệu quả:} margin = 0.5 là tham số tối ưu; RetinaFace đạt $>94\%$ precision ở điều kiện thông thường; cơ chế graceful degradation đảm bảo hệ thống không trả lỗi.
\end{itemize}

Chương tiếp theo sẽ tổng kết các đóng góp chính của luận văn và đề xuất hướng phát triển tương lai.
